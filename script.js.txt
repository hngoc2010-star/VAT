const priceM3=document.getElementById("priceM3");
const priceSheet=document.getElementById("priceSheet");

let editing="";

function num(v){
return Number(String(v).replace(/,/g,""))||0;
}

function money(v){
return Math.round(v).toLocaleString("en-US");
}

function calc(){

const thicknessBuy=
parseFloat(document.getElementById("thicknessBuy").value)||0;

const thicknessVat=
parseFloat(document.getElementById("thicknessVat").value)||0;

const width=
parseFloat(document.getElementById("width").value)||0;

const length=
parseFloat(document.getElementById("length").value)||0;

const bundles=
parseFloat(document.getElementById("bundles").value)||0;

const pcsBundle=
parseFloat(document.getElementById("pcsBundle").value)||0;

const pcs=bundles*pcsBundle;

const cbmBuy=
(thicknessBuy/1000)*
(width/1000)*
(length/1000)*
pcs;

const cbmVat=
(thicknessVat/1000)*
(width/1000)*
(length/1000)*
pcs;

const cbmSheet=
(thicknessVat/1000)*
(width/1000)*
(length/1000);

let pricePerM3=0;
let pricePerSheet=0;

if(editing==="m3"){

pricePerM3=num(priceM3.value);

pricePerSheet=
pricePerM3*cbmSheet;

if(pricePerM3>0){
priceSheet.value=money(pricePerSheet);
}

}else if(editing==="sheet"){

pricePerSheet=num(priceSheet.value);

pricePerM3=
pricePerSheet/cbmSheet;

if(pricePerSheet>0){
priceM3.value=money(pricePerM3);
}

}else{
return;
}

const vatPriceM3=
(pricePerM3*cbmBuy)/cbmVat;

const amount=
vatPriceM3*cbmVat;

const vat8=
amount*0.08;

const total=
amount+vat8;

document.getElementById("pcs").innerText=
money(pcs);

document.getElementById("cbmBuy").innerText=
cbmBuy.toFixed(2);

document.getElementById("cbmVat").innerText=
cbmVat.toFixed(2);

document.getElementById("vatPrice").innerText=
money(vatPriceM3);

document.getElementById("amount").innerText=
money(amount);

document.getElementById("vat8").innerText=
money(vat8);

document.getElementById("total").innerText=
money(total);
}

priceM3.addEventListener("input",()=>{

editing="m3";

priceSheet.value="";

calc();

});

priceSheet.addEventListener("input",()=>{

editing="sheet";

priceM3.value="";

calc();

});

document
.querySelectorAll("input")
.forEach(el=>{

if(el.id==="priceM3"||el.id==="priceSheet") return;

el.addEventListener("input",calc);

});
